function createList(title, items) { if (!items || !items.length) return ''; let html = `

${title}

    `; items.forEach(item => { if (typeof item === "string") { html += `
  • ${item}
    • `; } else if (typeof item === "object" && item.term && item.key_concept_desc) { html += `
  • ${item.term}: ${item.key_concept_desc}
    • `; } }); html += `
`; return html; }
